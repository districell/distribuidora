// Reglas Firestore (modo cerrado: solo admin autenticado)
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.email == 'admin@celldistribuciones.com';
    }
  }
}
// Reglas Storage (solo admin puede escribir)
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read;
      allow write: if request.auth != null && request.auth.token.email == 'admin@celldistribuciones.com';
    }
  }
}
